# CREATE TABLES

```sql id="b4x8mq"
-- =========================
-- MOVIE TABLE
-- =========================

CREATE TABLE Movie (
    Mname VARCHAR(30) PRIMARY KEY,
    release_year INT,
    budget DECIMAL(10,2)
);

-- =========================
-- ACTOR TABLE
-- =========================

CREATE TABLE Actor (
    A_name VARCHAR(30) PRIMARY KEY,
    role VARCHAR(30),
    charges DECIMAL(10,2),
    address VARCHAR(50),
    age INT
);

-- =========================
-- PRODUCER TABLE
-- =========================

CREATE TABLE Producer (
    producer_id INT PRIMARY KEY,
    pname VARCHAR(30),
    p_address VARCHAR(50)
);

-- =========================
-- MOVIE_ACTOR TABLE
-- =========================

CREATE TABLE Movie_Actor (
    Mname VARCHAR(30),
    A_name VARCHAR(30),

    PRIMARY KEY (Mname, A_name),

    FOREIGN KEY (Mname)
    REFERENCES Movie(Mname),

    FOREIGN KEY (A_name)
    REFERENCES Actor(A_name)
);

-- =========================
-- MOVIE_PRODUCER TABLE
-- =========================

CREATE TABLE Movie_Producer (
    Mname VARCHAR(30),
    producer_id INT,

    PRIMARY KEY (Mname, producer_id),

    FOREIGN KEY (Mname)
    REFERENCES Movie(Mname),

    FOREIGN KEY (producer_id)
    REFERENCES Producer(producer_id)
);
```

# 2) List names of producers of 'Chhava' movie

```sql id="w4z9qn"
SELECT p.pname
FROM Producer p
JOIN Movie_Producer mp
ON p.producer_id = mp.producer_id
WHERE mp.Mname = 'Chhava';
```

# Q2 SOLUTION

# 1) Trigger before inserting actor record

```sql id="tm8j2y"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION check_actor_age()
RETURNS TRIGGER AS
$$
BEGIN

    IF NEW.age < 0 OR NEW.age > 3 THEN
        RAISE EXCEPTION 'Invalid input';
    END IF;

    RETURN NEW;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="6qk1ps"
CREATE TRIGGER trg_check_actor_age
BEFORE INSERT
ON Actor
FOR EACH ROW
EXECUTE FUNCTION check_actor_age();
```

# 2) Stored Function to display movies released before year 2000

```sql id="cc9e2g"
CREATE OR REPLACE FUNCTION movies_before_2000()
RETURNS TABLE (
    movie_name VARCHAR,
    release_year INT,
    budget DECIMAL
)
AS
$$
BEGIN

    RETURN QUERY

    SELECT
        Mname,
        release_year,
        budget
    FROM Movie
    WHERE release_year < 2000;

END;
$$
LANGUAGE plpgsql;
```

# EXECUTE FUNCTION

