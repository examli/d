# CREATE TABLES

```sql id="l5m8qx"
-- =========================
-- AUTHOR TABLE
-- =========================

CREATE TABLE Author (
    aname VARCHAR(40) PRIMARY KEY,
    address VARCHAR(50),
    phone VARCHAR(15)
);

-- =========================
-- BOOK TABLE
-- =========================

CREATE TABLE Book (
    title VARCHAR(50) PRIMARY KEY,
    ISBN VARCHAR(20),
    category VARCHAR(30),
    aname VARCHAR(40),

    FOREIGN KEY (aname)
    REFERENCES Author(aname)
);
```

# Q2 SOLUTION

# 1) Trigger before inserting Book record

```sql id="w5t1qv"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION check_author_name()
RETURNS TRIGGER AS
$$
BEGIN

    IF NOT EXISTS (
        SELECT *
        FROM Author
        WHERE aname = NEW.aname
    ) THEN

        RAISE EXCEPTION 'Invalid Name';

    END IF;

    RETURN NEW;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="n4m8pe"
CREATE TRIGGER trg_check_author
BEFORE INSERT
ON Book
FOR EACH ROW
EXECUTE FUNCTION check_author_name();
```

# 2) Function using cursor to display all books of specific author

```sql id="x8j4lm"
CREATE OR REPLACE FUNCTION books_of_author(author_name VARCHAR)
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT title,
           category
    FROM Book
    WHERE aname = author_name;

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Book: %, Category: %',
        rec.title,
        rec.category;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```

---

# EXECUTE FUNCTION

```sql id="q7w2ds"

```
