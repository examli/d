# CREATE TABLES

```sql id="r8x4tm"
-- =========================
-- STUDENT TABLE
-- =========================

CREATE TABLE Student (
    sreg_no INT PRIMARY KEY,
    sname VARCHAR(30),
    class VARCHAR(10)
);

-- =========================
-- COMPETITION TABLE
-- =========================

CREATE TABLE Competition (
    c_no INT PRIMARY KEY,
    cname VARCHAR(30),
    c_type VARCHAR(30)
);

-- =========================
-- STUDENT_COMPETITION TABLE
-- =========================

CREATE TABLE Student_Competition (
    sreg_no INT,
    c_no INT,
    rank INT,
    year INT,

    PRIMARY KEY (sreg_no, c_no),

    FOREIGN KEY (sreg_no)
    REFERENCES Student(sreg_no),

    FOREIGN KEY (c_no)
    REFERENCES Competition(c_no)
);
```

# Q1 SOLUTION

# 1) List names of all students who participated in Quiz Competition

```sql id="f6q3yn"
SELECT s.sname
FROM Student s
JOIN Student_Competition sc
ON s.sreg_no = sc.sreg_no
JOIN Competition c
ON sc.c_no = c.c_no
WHERE c.cname = 'Quiz';
```

# Q2 SOLUTION

# 1) Trigger after insert on Student table

```sql id="d9r4yo"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION check_competition_count()
RETURNS TRIGGER AS
$$
DECLARE

    total INT;

BEGIN

    SELECT COUNT(*)
    INTO total
    FROM Student_Competition
    WHERE sreg_no = NEW.sreg_no;

    IF total > 2 THEN
        RAISE NOTICE 'Welcome for competition';
    END IF;

    RETURN NEW;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="q8t6mv"
CREATE TRIGGER trg_check_competition
AFTER INSERT
ON Student
FOR EACH ROW
EXECUTE FUNCTION check_competition_count();
```

# 2) Stored Function using Cursor

```sql id="u3p9ke"
CREATE OR REPLACE FUNCTION student_competition_details(regno INT)
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT s.sname,
           c.cname
    FROM Student s
    JOIN Student_Competition sc
    ON s.sreg_no = sc.sreg_no
    JOIN Competition c
    ON sc.c_no = c.c_no
    WHERE s.sreg_no = regno;

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Student: %, Competition: %',
        rec.sname,
        rec.cname;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```
