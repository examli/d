# CREATE TABLES

```sql id="t4m8xp"
-- =========================
-- OWNER TABLE
-- =========================

CREATE TABLE Owner (
    owner_id INT PRIMARY KEY,
    oname VARCHAR(30),
    address VARCHAR(50),
    phone VARCHAR(15)
);

-- =========================
-- PROPERTY TABLE
-- =========================

CREATE TABLE Property (
    pno INT PRIMARY KEY,
    description VARCHAR(50),
    area VARCHAR(30),
    owner_id INT,

    FOREIGN KEY (owner_id)
    REFERENCES Owner(owner_id)
);
```

# Q2 SOLUTION

# 1) Trigger before inserting description into Property table

```sql id="p9x5uk"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION check_property_area()
RETURNS TRIGGER AS
$$
BEGIN

    IF NEW.area <> 'Pune' THEN
        RAISE EXCEPTION 'Property area should be Pune';
    END IF;

    RETURN NEW;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="f7m2qy"
CREATE TRIGGER trg_check_property_area
BEFORE INSERT
ON Property
FOR EACH ROW
EXECUTE FUNCTION check_property_area();
```

# 2) Cursor to list details of all Owners

```sql id="q1w6ze"
CREATE OR REPLACE FUNCTION owner_cursor()
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT *
    FROM Owner;

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Owner ID: %, Name: %, Address: %, Phone: %',
        rec.owner_id,
        rec.oname,
        rec.address,
        rec.phone;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```

---

# EXECUTE FUNCTION

