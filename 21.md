# CREATE TABLES

```sql id="q5w8yn"
-- =========================
-- PRODUCER TABLE
-- =========================

CREATE TABLE Producer (
    producer_id INT PRIMARY KEY,
    pname VARCHAR(30),
    p_address VARCHAR(30)
);

-- =========================
-- MOVIE TABLE
-- =========================

CREATE TABLE Movie (
    mid INT PRIMARY KEY,
    mname VARCHAR(25),
    release_year INT,
    budget MONEY,
    producer_id INT,

    FOREIGN KEY (producer_id)
    REFERENCES Producer(producer_id)
);

-- =========================
-- ACTOR TABLE
-- =========================

CREATE TABLE Actor (
    a_name VARCHAR(30) PRIMARY KEY,
    city VARCHAR(30),
    charges MONEY,
    mid INT,

    FOREIGN KEY (mid)
    REFERENCES Movie(mid)
);
```

# Q2 SOLUTION

# 1) Trigger before inserting record into Movie table

```sql id="k3m8uv"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION check_release_year()
RETURNS TRIGGER AS
$$
BEGIN

    IF NEW.release_year > EXTRACT(YEAR FROM CURRENT_DATE) THEN
        RAISE EXCEPTION 'Release year should not be greater than current year';
    END IF;

    RETURN NEW;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="r6t1qx"
CREATE TRIGGER trg_check_release_year
BEFORE INSERT
ON Movie
FOR EACH ROW
EXECUTE FUNCTION check_release_year();
```

# 2) Cursor using function to list movie-wise charges of 'Amitabh Bachchan'

```sql id="f8v2my"
CREATE OR REPLACE FUNCTION movie_actor_charges()
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT m.mname,
           a.charges
    FROM Movie m
    JOIN Actor a
    ON m.mid = a.mid
    WHERE a.a_name = 'Amitabh Bachchan';

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Movie: %, Charges: %',
        rec.mname,
        rec.charges;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```
