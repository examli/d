# CREATE TABLES

```sql id="m8x4tp"
-- =========================
-- DEPARTMENT TABLE
-- =========================

CREATE TABLE Dept (
    dept_no VARCHAR(10) PRIMARY KEY,
    dept_name VARCHAR(20)
);

-- =========================
-- SALESMAN TABLE
-- =========================

CREATE TABLE Salesman (
    sno INT PRIMARY KEY,
    s_name VARCHAR(30),
    start_year INT,
    dept_no VARCHAR(10),

    FOREIGN KEY (dept_no)
    REFERENCES Dept(dept_no)
);

-- =========================
-- TRIP TABLE
-- =========================

CREATE TABLE Trip (
    tno INT PRIMARY KEY,
    from_city VARCHAR(20),
    to_city VARCHAR(20),
    departure_date DATE,
    return_date DATE,
    sno INT,

    FOREIGN KEY (sno)
    REFERENCES Salesman(sno)
);

-- =========================
-- EXPENSE TABLE
-- =========================

CREATE TABLE Expense (
    eid INT PRIMARY KEY,
    amount MONEY,
    tno INT,

    FOREIGN KEY (tno)
    REFERENCES Trip(tno)
);
```

# Q2 SOLUTION

# 1) Cursor to display trip details having maximum expense

```sql id="p1v8xn"
CREATE OR REPLACE FUNCTION max_expense_trip()
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT t.*
    FROM Trip t
    JOIN Expense e
    ON t.tno = e.tno
    WHERE e.amount = (
        SELECT MAX(amount)
        FROM Expense
    );

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Trip No: %, From: %, To: %',
        rec.tno,
        rec.from_city,
        rec.to_city;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```


# 2) Stored Function to count business trips from Pune to Mumbai

```sql id="f2m9wk"
CREATE OR REPLACE FUNCTION count_pune_mumbai_trips()
RETURNS INT AS
$$
DECLARE

    total INT;

BEGIN

    SELECT COUNT(*)
    INTO total
    FROM Trip
    WHERE from_city = 'Pune'
    AND to_city = 'Mumbai';

    RETURN total;

END;
$$
LANGUAGE plpgsql;
```

---

# EXECUTE FUNCTION

