# CREATE TABLES

```sql id="m7x2tp"
-- =========================
-- PRODUCER TABLE
-- =========================

CREATE TABLE Producer (
    producer_id INT PRIMARY KEY,
    pname VARCHAR(30),
    p_address VARCHAR(30)
);

-- =========================
-- MOVIE TABLE
-- =========================

CREATE TABLE Movie (
    m_name VARCHAR(25) PRIMARY KEY,
    release_year INT,
    budget MONEY,
    producer_id INT,

    FOREIGN KEY (producer_id)
    REFERENCES Producer(producer_id)
);

-- =========================
-- ACTOR TABLE
-- =========================

CREATE TABLE Actor (
    a_name VARCHAR(30),
    city VARCHAR(30),
    charges MONEY
);

-- =========================
-- MOVIE_ACTOR TABLE
-- =========================

CREATE TABLE Movie_Actor (
    m_name VARCHAR(25),
    a_name VARCHAR(30),

    PRIMARY KEY (m_name, a_name),

    FOREIGN KEY (m_name)
    REFERENCES Movie(m_name)
);
```

# Q2 SOLUTION

# 1) Trigger before inserting record into Movie table

```sql id="n4x7qr"
-- =========================
-- TRIGGER FUNCTION
-- =========================

CREATE OR REPLACE FUNCTION check_release_year()
RETURNS TRIGGER AS
$$
BEGIN

    IF NEW.release_year > EXTRACT(YEAR FROM CURRENT_DATE) THEN
        RAISE EXCEPTION 'Release year should not be greater than current year';
    END IF;

    RETURN NEW;

END;
$$
LANGUAGE plpgsql;
```

---

# CREATE TRIGGER

```sql id="u7m5tp"
CREATE TRIGGER trg_check_release_year
BEFORE INSERT
ON Movie
FOR EACH ROW
EXECUTE FUNCTION check_release_year();
```

# 2) Cursor using function to list movie-wise charges of 'Madhuri Dixit'

```sql id="z8r3wn"
CREATE OR REPLACE FUNCTION madhuri_movie_charges()
RETURNS VOID AS
$$
DECLARE

    rec RECORD;

    cur CURSOR FOR
    SELECT m.m_name,
           a.charges
    FROM Movie m
    JOIN Movie_Actor ma
    ON m.m_name = ma.m_name
    JOIN Actor a
    ON ma.a_name = a.a_name
    WHERE a.a_name = 'Madhuri Dixit';

BEGIN

    OPEN cur;

    LOOP

        FETCH cur INTO rec;

        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'Movie: %, Charges: %',
        rec.m_name,
        rec.charges;

    END LOOP;

    CLOSE cur;

END;
$$
LANGUAGE plpgsql;
```
