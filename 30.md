# CREATE TABLES

```sql id="x7m3tp"
-- =========================
-- BRANCH TABLE
-- =========================

CREATE TABLE Branch (
    bid INT PRIMARY KEY,
    brname VARCHAR(30),
    brcity VARCHAR(10)
);

-- =========================
-- CUSTOMER TABLE
-- =========================

CREATE TABLE Customer (
    cno INT PRIMARY KEY,
    cname VARCHAR(20),
    caddr VARCHAR(35),
    city VARCHAR(20)
);

-- =========================
-- LOAN_APPLICATION TABLE
-- =========================

CREATE TABLE Loan_Application (
    lno INT PRIMARY KEY,
    lamtrequired MONEY,
    lamtapproved MONEY,
    l_date DATE,
    bid INT,
    cno INT,

    FOREIGN KEY (bid)
    REFERENCES Branch(bid),

    FOREIGN KEY (cno)
    REFERENCES Customer(cno)
);
```

# Q1 SOLUTION

# 1) List names of customers who received loan amount 1 lakh

```sql id="m8t2xp"
SELECT c.cname
FROM Customer c
JOIN Loan_Application l
ON c.cno = l.cno
WHERE l.lamtapproved = 100000;
```

---

# 2) Find maximum loan amount approved

```sql id="r5v9wn"
SELECT MAX(lamtapproved) AS maximum_loan_approved
FROM Loan_Application;
```

---

# 3) Find total loan amount sanctioned by 'Deccan' branch

```sql id="k1q7tx"
SELECT SUM(l.lamtapproved) AS total_loan_amount
FROM Loan_Application l
JOIN Branch b
ON l.bid = b.bid
WHERE b.brname = 'Deccan';
```

---

# 4) Count loan applications received by 'J.M.Road' branch

```sql id="p6w4vm"
SELECT COUNT(*) AS total_applications
FROM Loan_Application l
JOIN Branch b
ON l.bid = b.bid
WHERE b.brname = 'J.M.Road';
```

# Q2 SOLUTION

# 1) Create View to display customers who received loan amount less than requirement

```sql id="u2x8rp"
CREATE VIEW customer_loan_view AS
SELECT c.*,
       l.lamtrequired,
       l.lamtapproved
FROM Customer c
JOIN Loan_Application l
ON c.cno = l.cno
WHERE l.lamtapproved < l.lamtrequired;
```

---

# DISPLAY VIEW


# 2) Stored Function to display customers who applied in more than one branch

```sql id="t4q9xn"
CREATE OR REPLACE FUNCTION multiple_branch_customers()
RETURNS TABLE (
    customer_name VARCHAR,
    city_name VARCHAR,
    total_branches BIGINT
)
AS
$$
BEGIN

    RETURN QUERY

    SELECT
        c.cname,
        c.city,
        COUNT(DISTINCT l.bid)

    FROM Customer c
    JOIN Loan_Application l
    ON c.cno = l.cno

    GROUP BY c.cname, c.city

    HAVING COUNT(DISTINCT l.bid) > 1;

END;
$$
LANGUAGE plpgsql;
```
